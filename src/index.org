#+begin_src emacs-lisp :exports results :results html
  (let ((posts-sorted (sort (copy-sequence blog/all-posts) ;; my/posts is global variable with metadata of all the posts.
                            (lambda (a b) (time-less-p (car b) (car a))))))
    (mapconcat
     (lambda (p)
       (let ((date-str (format-time-string "%Y-%m-%d" (nth 0 p)))
             (title (nth 1 p))
             (filename (nth 2 p)))
         (format (concat "<a class=\"post-entry\" href=\"./posts/%s\">"
                         "  <time datetime=\"%s\">%s</time>"
                         "  <b class=\"post-entry-title\">%s</b>"
                         "</a>")
                 (string-replace ".org" ".html" filename)
                 date-str
                 date-str
                 title)))
     posts-sorted
     "\n"))
#+end_src
